Exercise 2: Using Math

						
							function setup() {
								createCanvas(600, 200);
							}

							function draw() {
								background(220);

								translate(width / 2, height / 2); // move to center of canvas
								translate(-20, -20); // adjust for offset of top left corner of figure
								translate(-35, -35); // move by half width and height of figure, so positioning is relative to center of figure

								rect(50, 20, 10, 10);
								rect(20, 30, 70, 60);
								rect(40, 70, 30, 10);
								circle(40, 50, 20);
								circle(70, 50, 20);
							}

Exercise 2x: Using Math

						
							function setup() {
								createCanvas(600, 200);
							}

							function draw() {
								background(220);
								
								let horizontalAnchor = 0.1;
								let verticalAnchor = 0.9;

								translate(width * horizontalAnchor, height * verticalAnchor); // move to relative position on canvas specified by horizontalAnchor and verticalAnchor
								translate(-20, -20); // adjust for offset of top left corner of figure
								translate(-35, -35); // move by half width and height of figure, so positioning is relative to center of figure

								rect(50, 20, 10, 10);
								rect(20, 30, 70, 60);
								rect(40, 70, 30, 10);
								circle(40, 50, 20);
								circle(70, 50, 20);
							}
