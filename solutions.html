Exercise 12x: Interactive DOM

						
							<html>
								<head>
									<style>
										.todoItem {
											display: block;
										}
									</style>
								</head>
								<body>
									<h1>Todos</h1>
									<input name="newItem" id="newItem"></input>
									<button id="addItem">Hinzufügen</button>
									<div id="todoList"></div>
									<h1>Done</h1>
									<div id="doneList"></div>
									
									<script>
										const todoList = document.getElementById('todoList');
										const doneList = document.getElementById('doneList');
										const addItemButton = document.getElementById('addItem');
										const newItemText = document.getElementById('newItem');
										
										const todoItems = [];
										
										function renderTodoItem (todoItem) {
											todoItems.push(todoItem);
											
											const todoItemElement = document.createElement('label');
											todoItemElement.classList.add('todoItem');
											
											const checkbox = document.createElement('input');
											checkbox.setAttribute('type', 'checkbox');
											checkbox.checked = todoItem.done;
											checkbox.addEventListener('change', () => {
												todoItem.done = checkbox.checked;
												localStorage.setItem('savedTodoItems', JSON.stringify(todoItems));
												
												if (checkbox.checked) {
													todoList.removeChild(todoItemElement);
													doneList.appendChild(todoItemElement);
												} else {
													doneList.removeChild(todoItemElement);
													todoList.appendChild(todoItemElement);
												}
											});
											
											todoItemElement.appendChild(checkbox);
											
											const label = document.createElement('span');
											label.innerHTML = todoItem.text;

											todoItemElement.appendChild(label);
											
											if (todoItem.done) {
												doneList.appendChild(todoItemElement);
											} else {
												todoList.appendChild(todoItemElement);
											}
										}
										
										addItemButton.addEventListener('click', () => {
											const text = newItemText.value;
											const todoItem = {
												text,
												done: false
											}
											
											renderTodoItem(todoItem);
											
											localStorage.setItem('savedTodoItems', JSON.stringify(todoItems));
											
											newItemText.value = '';
										});
										
										const savedTodoItems = localStorage.getItem('savedTodoItems');
										
										if (savedTodoItems) {
											const savedTodoItemsList = JSON.parse(savedTodoItems);
											
											for (const todoItem of savedTodoItemsList) {
												renderTodoItem(todoItem);
											}
										}
									</script>
								</body>
							</html>

Exercise 15: Loops

						
							<html>
								<head>
									<style>
										label {
											display: block;
										}
									</style>
								</head>
								<body>
									<label for="series1"><input type="radio" name="series" id="series1"> F<sub>n</sub> = n * n</label></input>
									<label for="series2"><input type="radio" name="series" id="series2"> F<sub>n</sub> = F<sub>n-1</sub> * 2</label></input>
									<label for="series3"><input type="radio" name="series" id="series3"> F<sub>n</sub> = F<sub>n-1</sub> + F<sub>n-2</sub></label></input>
									<label for="series4"><input type="radio" name="series" id="series4"> F<sub>n</sub> = nth prime number</label></input>
									<label for="numberOfElements">Number of Elements: <input name="numberOfElements" id="numberOfElements"></input></label>
									<button id="calculate">Calculate</button>
									<div id="output"></div>
									
									<script>
										const seriesRadioButtons = []
										
										for (let i = 1; i <= 4; i++) {
											seriesRadioButtons.push(document.getElementById('series' + i));
										}
										
										function getActiveRadioButton (radioButtons) {
											for (const button of radioButtons) {
												if (button.checked) {
													return button.getAttribute('id');
												}
											}
											
											return null;
										}
										
										function isPrime (n) {
											for (let i = 2; i * i < n; i += i < 3 ? 1 : 2) {
												if (n % i === 0) {
													return false;
												}
											}
											
											return true;
										}
										
										const numberOfElementsInput = document.getElementById('numberOfElements');
										const output = document.getElementById('output');
										const calculateButton = document.getElementById('calculate');
										
										calculateButton.addEventListener('click', () => {
											const selectedSeries = getActiveRadioButton(seriesRadioButtons);
											const numberOfElements = parseInt(numberOfElementsInput.value, 10);
											let elements = []
											
											if (!Number.isNaN(numberOfElements) && selectedSeries !== null) {
												let lastPrime = 1;
												
												for (let i = 0; i < numberOfElements; i++) {
													let n = i + 1;
													let element = null
													
													if (selectedSeries === 'series1') {
														element = n * n;
													} else if (selectedSeries === 'series2') {
														if (i === 0) {
															element = 1;
														} else {
															element = elements[i - 1] * 2;
														}
													} else if (selectedSeries === 'series3') {
														if (i < 2) {
															element = 1;
														} else {
															element = elements[i - 2] + elements[i - 1];
														}
													} else if (selectedSeries === 'series4') {
														do {
															lastPrime += lastPrime < 3 ? 1 : 2;
														} while (!isPrime(lastPrime));
														
														element = lastPrime;
													}
													
													if (element !== null) {
														elements.push(element);
													}
												}
											}

							function swap (array, i, j) {
								if (
									array.hasOwnProperty(i)
									&& array.hasOwnProperty(j)
								) {
									const tmp = array[i];
									array[i] = array[j];
									array[j] = tmp;
								} else if (array.hasOwnProperty(i)) {
									array[j] = array[i];
									delete array[i];
								} else if (array.hasOwnProperty(j)) {
									array[i] = array[j];
									delete array[j];
								}
							}
