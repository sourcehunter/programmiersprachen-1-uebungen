Without running the code: What is the result?

						
							function calculate (mode, a, b) {
								if (mode === '+') {
									return a + b;
								} else if (mode === '*') {
									return a * b;
								} else if (mode === '/') {
									return a / b;
								} else if (mode === '-') {
									return a - b;
								} else if (mode === '^') {
									return Math.pow(a, b);
								} else {
									return Number.NaN;
								}
							}

Exercise 13x: Understanding Code

Without running the code: What is the result?

						
							function factors (value) {
								let rem = value;
								let factors = [];
								
								let i = 2;
								
								while (i <= rem) {
									while (rem % i === 0) {
										factors.push(i);
										rem /= i;
									}
									
									i += i === 2 ? 1 : 2;
								}
								
								return factors;
							}
